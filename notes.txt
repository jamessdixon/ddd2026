REBALANCE THE APP SERVERS 
az vm run-command invoke -g rg-jxd-ddd --name flowdemo-applb \
  --command-id RunShellScript \
  --scripts "
    cat > /etc/nginx/sites-available/appproxy << 'EOF'
upstream appvms {
    server 10.0.2.10:8080 weight=50;
    server 10.0.2.11:8080 weight=30;
    server 10.0.2.12:8080 weight=10;
    server 10.0.2.13:8080 weight=10;
}
server {
    listen 80;
    location / {
        proxy_pass http://appvms;
        proxy_set_header Host \$host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
    }
    location /health {
        return 200 'app-lb-ok';
        add_header Content-Type text/plain;
    }
}
EOF
    nginx -t && systemctl reload nginx
    echo 'Done'
  "

# Part 1 — Add explicit deny rule in app-nsg for inter-app-vm traffic
az network nsg rule create -g rg-jxd-ddd --nsg-name flowdemo-app-nsg \
  --name Deny-AppVM-To-AppVM \
  --priority 150 \
  --direction Inbound \
  --source-address-prefixes 10.0.2.0/24 \
  --destination-address-prefixes 10.0.2.0/24 \
  --destination-port-ranges 8080 \
  --protocol Tcp \
  --access Deny


#Part 2 - set up the chatty server
az vm run-command invoke -g rg-jxd-ddd --name flowdemo-app4 \
  --command-id RunShellScript \
  --scripts "
    printf '#!/bin/bash\nTIMESTAMP=\$(date -u +%%Y-%%m-%%dT%%H:%%M:%%SZ)\necho \"\${TIMESTAMP} - app4 attempting outbound calls\"\nfor ip in 10.0.2.10 10.0.2.11 10.0.2.12; do\n  echo -n \"  Calling \${ip}:8080 ... \"\n  curl -s --connect-timeout 5 http://\${ip}:8080/health && echo OK || echo DENIED/TIMEOUT\ndone\n' > /opt/chatty.sh
    chmod +x /opt/chatty.sh
    head -3 /opt/chatty.sh
    systemctl start chatty.service
    sleep 2
    journalctl -u chatty.service -n 10 --no-pager
  "
